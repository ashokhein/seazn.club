```ts
const STAGE_DECIDER_KEYS = ["shootout", "extraTime"] as const;

/** Overlay knockout-decider keys from the stage config onto the division sport
 *  config (PROMPT-61 §2). Identity when the stage sets neither key. */
export function stageScopedCfg(
  divisionCfg: unknown,
  stageCfg: Record<string, unknown> | null | undefined,
): unknown {
  if (stageCfg == null) return divisionCfg;
  const overlay: Record<string, unknown> = {};
  for (const key of STAGE_DECIDER_KEYS) {
    if (stageCfg[key] !== undefined) overlay[key] = stageCfg[key];
  }
  if (Object.keys(overlay).length === 0) return divisionCfg;
  return { ...(divisionCfg as Record<string, unknown>), ...overlay };
}
```

```ts
const normPool = (s: string) => s.trim().toLowerCase().replace(/^pool\s+/, "");
function poolByName(tables: StageTables, name: string): PoolTable {
```

```ts
const ADDABLE = new Set(["league", "group", "swiss"]);

export async function addFixture(
  auth: AuthCtx,
  stageId: string,
  input: { home_entrant_id: string; away_entrant_id: string; round_no?: number; scheduled_at?: string | null; venue?: string | null },
): Promise<{ fixture_id: string }> {
  return withTenant(auth.orgId, async (tx) => {
    const [stage] = await tx<{ division_id: string; kind: string; status: string }[]>`
      select division_id, kind, status from stages where id = ${stageId}`;
    if (!stage) throw new HttpError(404, "stage not found");
    if (stage.kind === "ladder") throw new HttpError(422, "use challenges to add a ladder match");
    if (stage.kind === "americano") throw new HttpError(422, "americano rounds are generated — add another round instead");
    if (!ADDABLE.has(stage.kind)) {
      throw new HttpError(422, "can't add a loose fixture to a bracket — it has no slot in the tree");
    }
    if (stage.status === "complete") {
      throw new HttpError(422, "this stage is complete — reopen it before adding a match");
    }
    if (input.home_entrant_id === input.away_entrant_id) throw new HttpError(422, "an entrant cannot play itself");
    await tx`select pg_advisory_xact_lock(hashtext(${"division:" + stage.division_id}))`;
    const entrants = await tx<{ id: string }[]>`
      select id from entrants where division_id = ${stage.division_id}
        and id in (${input.home_entrant_id}, ${input.away_entrant_id})`;
    if (entrants.length !== 2) throw new HttpError(422, "both entrants must belong to this division");

    // stage fixture involving them (no separate membership table needed).
```
